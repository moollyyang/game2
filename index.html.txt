const questions = [

"想喝好幾種飲料，結果每杯都喝一點，常常沒喝完。",
"看影片時，會同時滑手機、回訊息或做其他事情。",
"別人講話時，常聽到一半就開始想別的事情。",
"做事情做到一半，突然想到別的事就跑去做。",
"明明剛放下鑰匙、手機或錢包，卻馬上找不到。",

"房間、桌面或包包容易累積雜物，很難維持整齊。",
"用過的衛生紙、杯子或包裝常隨手放著忘記收。",
"常忘記繳費、回覆訊息、預約或其他待辦事項。",
"明知道有重要事情要做，卻一直拖到最後一刻。",
"安排行程時，經常低估需要花費的時間。",

"購物時容易衝動下單，之後才發現其實不需要。",
"別人話還沒講完，就忍不住插話或直接回答。",
"看到感興趣的東西，會立刻想研究或購買。",

"即使坐著，也常不自覺抖腳、轉筆或摸東西。",
"腦中經常同時跑很多想法，很難真正放空休息。"

];

const container = document.getElementById("questions");

questions.forEach((q,index)=>{

let html = `
<div class="question">
<p>${index+1}. ${q}</p>

<div class="options">

<label>
<input type="radio" name="q${index}" value="0" required>
0 幾乎沒有
</label>

<label>
<input type="radio" name="q${index}" value="1">
1 偶爾
</label>

<label>
<input type="radio" name="q${index}" value="2">
2 經常
</label>

<label>
<input type="radio" name="q${index}" value="3">
3 幾乎每天
</label>

</div>
</div>
`;

container.innerHTML += html;

});

function calculateScore(){

let total = 0;

for(let i=0;i<questions.length;i++){

const selected = document.querySelector(`input[name="q${i}"]:checked`);

if(!selected){
alert("請完成所有題目");
return;
}

total += Number(selected.value);
}

let resultText = "";
let resultClass = "";

if(total <= 10){
resultText = "ADHD 傾向較低";
resultClass = "low";
}
else if(total <= 20){
resultText = "有部分 ADHD 特徵";
resultClass = "mid";
}
else if(total <= 30){
resultText = "中度 ADHD 傾向，建議進一步了解";
resultClass = "high";
}
else{
resultText = "較高 ADHD 傾向，可考慮尋求專業評估";
resultClass = "very-high";
}

const result = document.getElementById("result");

result.style.display = "block";
result.className = `result ${resultClass}`;

result.innerHTML = `
<h2>檢測結果</h2>
<p><strong>總分：${total} / 45</strong></p>
<p>${resultText}</p>
<p style="margin-top:15px;color:#666;">
本測驗僅供自我了解與篩檢參考，
無法作為 ADHD 診斷依據。
若症狀已影響工作、人際或生活功能，
建議諮詢精神科醫師或臨床心理師。
</p>
`;

window.scrollTo({
top: document.body.scrollHeight,
behavior:"smooth"
});

}
